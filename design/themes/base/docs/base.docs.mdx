import { Button, Box, ChakraProvider, extendTheme } from '@chakra-ui/react';

An example of chakra UI theme extension.

Recommend to use use `CustomThemeTypings` to define theme object.

```ts
import type { CustomThemeTypings } from '@chakra-ui/react';

// This type will ensure that your theme is compatible with the Chakra UI theme provider
export const myTheme: CustomThemeTypings = {
  semanticTokens: {},
};
```

## Try it out!

```tsx live
() => {
  // Customize this theme to see how it affects the components
  const theme = {
    semanticTokens: {
      colors: {
        error: 'red.500',
        success: 'green.500',
        primary: {
          default: 'blue.500',
          _dark: 'blue.400',
        },
        secondary: {
          default: 'yellow.800',
          _dark: 'yellow.700',
        },
      },
    },
  };

  return (
    <ChakraProvider theme={extendTheme(theme)}>
      <Box
        gap={4}
        display='flex'
      >
        <Button bg='primary'>Primary</Button>
        <Button bg='secondary'>Secondary</Button>
        <Button bg='error'>Error</Button>
        <Button bg='success'>Success</Button>
      </Box>
    </ChakraProvider>
  );
};
```
